GLAVNI  
           ;UCITAVANJE ADRESE GPIO-A U S1(SAVED REGISTER)
           LUI S1, %HI(0XFFFF0F00)
           ADDI S1, S1, %LO(0XFFFF0F00)

           ADDI S2, S2, %LO(PODATAK)

           ;UCITAVANJE ADRESE ZNAMENAKA
           LUI S3, %HI(ZNAMENKE)
           ADDI S3, S3, %LO(ZNAMENKE)

           ;UCITAJ 100 U S4
           LW S4, 32(S3)

           ;AKO JE A0(UCITANI PODATAK) MANJI DO S4(100), PROVJERI JE LI U PITANJU ZNAMENKA 90
           BLT A0, S4, ZNM90

           ;AKO CITANI PODATAK NIJE MANJI OD 100, ONDA JE SIGURNO 100 JER ZADATKOM NE MOZE BITI VECI BROJ
           ;SPREMAM ODGOVARAJUCU ASCII VRIJEDNOST I REGISTAR A0 I POZOVI FJU ZA ISPIS
           ADDI A1, ZERO, 0X43;OX43 -> C (100)
           JAL RA, LCDWR

           ;AZURIRAJ PODATAK
           ADDI A0, A0, -100
           
           ;PROGRAM JE GOTOV
           JAL RA, KRAJ
